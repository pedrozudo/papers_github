% \edge [options] {inputs} {outputs}
\newcommand{\edge}[3][]{ %
  % Connect all nodes #2 to all nodes #3.
  \foreach \x in {#2} { %
    \foreach \y in {#3} { %
      \path (\x) edge [-,line width=0.6pt,#1] (\y) ;%
    } ;
  } ;
}

\newcommand{\upedge}[3][]{
  \path (#2) edge [line width=1pt,->,>=stealth,#1] (#3);
}

\newcommand{\ledge}[4][]{
